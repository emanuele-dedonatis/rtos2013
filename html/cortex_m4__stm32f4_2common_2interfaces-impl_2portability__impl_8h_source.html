40 /*
41  * This pointer is used by the kernel, and should not be used by end users.
42  * this is a pointer to a location where to store the thread's registers during
43  * context switch. It requires C linkage to be used inside asm statement.
44  * Registers are saved in the following order:
45  * *ctxsave+100 --> s31
46  * ...
47  * *ctxsave+40 --> s16
48  * *ctxsave+36 --> lr (contains EXC_RETURN whose bit #4 tells if fpu is used)
49  * *ctxsave+32 --> r11
50  * *ctxsave+28 --> r10
51  * *ctxsave+24 --> r9
52  * *ctxsave+20 --> r8
53  * *ctxsave+16 --> r7
54  * *ctxsave+12 --> r6
55  * *ctxsave+8 --> r5
56  * *ctxsave+4 --> r4
57  * *ctxsave+0 --> psp
58  */
59 extern "C" {
60 extern volatile unsigned int *ctxsave;
61 }
62 
70 #define saveContext() \
71 { \
72  asm volatile(" mrs r1, psp \n"/*get PROCESS stack ptr */ \
73  " ldr r0, =ctxsave \n"/*get current context */ \
74  " ldr r0, [r0] \n" \
75  " stmia r0!, {r1,r4-r11,lr} \n"/*save r1(psp),r4-r11,lr */ \
76  " lsls r2, lr, #27 \n"/*check if bit #4 is set */ \
77  " bmi 0f \n" \
78  " vstmia.32 r0, {s16-s31} \n"/*save s16-s31 if we need*/ \
79  "0: \n" \
80  ); \
81 }
82 
89 #define restoreContext() \
90 { \
91  asm volatile(" ldr r0, =ctxsave \n"/*get current context */ \
92  " ldr r0, [r0] \n" \
93  " ldmia r0!, {r1,r4-r11,lr} \n"/*load r1(psp),r4-r11,lr */ \
94  " lsls r2, lr, #27 \n"/*check if bit #4 is set */ \
95  " bmi 0f \n" \
96  " vldmia.32 r0, {s16-s31} \n"/*restore s16-s31 if need*/ \
97  "0: msr psp, r1 \n"/*restore PROCESS sp*/ \
98  " bx lr \n"/*return*/ \
99  ); \
100 }

118 inline void doDisableInterrupts()
119 {
120  // Documentation says __disable_irq() disables all interrupts with
121  // configurable priority, so also SysTick and SVC.
122  // No need to disable faults with __disable_fault_irq()
123  __disable_irq();
124  //The new fastDisableInterrupts/fastEnableInterrupts are inline, so there's
125  //the need for a memory barrier to avoid aggressive reordering
126  asm volatile("":::"memory");
127 }
128 
129 inline void doEnableInterrupts()
130 {
131  __enable_irq();
132  //The new fastDisableInterrupts/fastEnableInterrupts are inline, so there's
133  //the need for a memory barrier to avoid aggressive reordering
134  asm volatile("":::"memory");
135 }

137 inline bool checkAreInterruptsEnabled()
138 {
139  register int i;
140  asm volatile("mrs %0, primask \n\t":"=r"(i));
141  if(i!=0) return false;
142  return true;
143 }
