55  class TypeDependentOperation
56  {
57  public:
64  static void operation(int32_t *a, const int32_t *b, Op op)
65  {
66  T *o1=reinterpret_cast<T*>(a);
67  const T *o2=reinterpret_cast<const T*>(b);
68  switch(op)
69  {
70  case CALL:
71  (*o1)();
72  break;
73  case COPY:
74  *o1=*o2;
75  break;
76  case DESTROY:
77  o1->~T();
78  break;
79  }
80  }
