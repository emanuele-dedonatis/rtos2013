168  static void IRQrunRegulator(bool allReadyThreadsSaturated)
169  {
170  using namespace std;
171  #ifdef SCHED_CONTROL_FIXED_POINT
172  //The fixed point scheduler may overflow if Tr is higher than this
173  Tr=min(Tr,524287);
174  #endif //FIXED_POINT_MATH
175  #ifdef ENABLE_REGULATOR_REINIT
176  if(reinitRegulator==false)
177  {
178  #endif //ENABLE_REGULATOR_REINIT
179  int eTr=SP_Tr-Tr;
180  #ifndef SCHED_CONTROL_FIXED_POINT
181  int bc=bco+static_cast<int>(krr*eTr-krr*zrr*eTro);
182  #else //FIXED_POINT_MATH
183  //Tr is clamped to 524287, so eTr uses at most 19bits. Considering
184  //the 31bits of a signed int, we have 12bits free.
185  const int fixedKrr=static_cast<int>(krr*2048);
186  const int fixedKrrZrr=static_cast<int>(krr*zrr*1024);
187  int bc=bco+(fixedKrr*eTr)/2048-(fixedKrrZrr*eTro)/1024;
188  #endif //FIXED_POINT_MATH
189  if(allReadyThreadsSaturated)
190  {
191  //If all inner regulators reached upper saturation,
192  //allow only a decrease in the burst correction.
193  if(bc<bco) bco=bc;
194  } else bco=bc;
195 
196  bco=min<int>(max(bco,-Tr),bMax*threadListSize);
197  #ifndef SCHED_CONTROL_FIXED_POINT
198  float nextRoundTime=static_cast<float>(Tr+bco);
199  #else //FIXED_POINT_MATH
200  unsigned int nextRoundTime=Tr+bco; //Bounded to 20bits
201  #endif //FIXED_POINT_MATH
202  eTro=eTr;
203  Tr=0;//Reset round time
204  for(Thread *it=threadList;it!=0;it=it->schedData.next)
205  {
206  //Recalculate per thread set point
207  #ifndef SCHED_CONTROL_FIXED_POINT
208  it->schedData.SP_Tp=static_cast<int>(
209  it->schedData.alfa*nextRoundTime);
210  #else //FIXED_POINT_MATH
211  //nextRoundTime is bounded to 20bits, alfa to 12bits,
212  //so the multiplication fits in 32bits
213  it->schedData.SP_Tp=(it->schedData.alfa*nextRoundTime)/4096;
214  #endif //FIXED_POINT_MATH
215 
216  //Run each thread internal regulator
217  int eTp=it->schedData.SP_Tp - it->schedData.Tp;
218  //note: since b and bo contain the real value multiplied by
219  //multFactor, this equals b=bo+eTp/multFactor.
220  int b=it->schedData.bo + eTp;
221  //saturation
222  it->schedData.bo=min(max(b,bMin*multFactor),bMax*multFactor);
223  }
224  #ifdef ENABLE_REGULATOR_REINIT
225  } else {
226  reinitRegulator=false;
227  Tr=0;//Reset round time
228  //Reset state of the external regulator
229  eTro=0;
230  bco=0;
231 
232  for(Thread *it=threadList;it!=0;it=it->schedData.next)
233  {
234  //Recalculate per thread set point
235  #ifndef SCHED_CONTROL_FIXED_POINT
236  it->schedData.SP_Tp=static_cast<int>(it->schedData.alfa*SP_Tr);
237  #else //FIXED_POINT_MATH
238  //SP_Tr is bounded to 20bits, alfa to 12bits,
239  //so the multiplication fits in 32bits
240  it->schedData.SP_Tp=(it->schedData.alfa*SP_Tr)/4096;
241  #endif //FIXED_POINT_MATH
242 
243  int b=it->schedData.SP_Tp*multFactor;
244  it->schedData.bo=min(max(b,bMin*multFactor),bMax*multFactor);
245  }
246  }
247  #endif //ENABLE_REGULATOR_REINIT
248  }
