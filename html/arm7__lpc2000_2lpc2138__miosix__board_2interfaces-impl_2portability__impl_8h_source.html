39 /*
40  * This pointer is used by the kernel, and should not be used by end users.
41  * this is a pointer to a location where to store the thread's registers during
42  * context switch. It requires C linkage to be used inside asm statement.
43  * Registers are saved in the following order:
44  * *ctxsave+64 --> cpsr
45  * *ctxsave+60 --> pc (return address)
46  * *ctxsave+56 --> lr
47  * *ctxsave+52 --> sp
48  * *ctxsave+48 --> r12
49  * *ctxsave+44 --> r11
50  * *ctxsave+40 --> r10
51  * *ctxsave+36 --> r9
52  * *ctxsave+32 --> r8
53  * *ctxsave+28 --> r7
54  * *ctxsave+24 --> r6
55  * *ctxsave+20 --> r5
56  * *ctxsave+16 --> r4
57  * *ctxsave+12 --> r3
58  * *ctxsave+8 --> r2
59  * *ctxsave+4 --> r1
60  * *ctxsave+0 --> r0
61  */
62 extern "C" {
63 extern volatile unsigned int *ctxsave;
64 }

72 #define saveContextFromSwi() \
73 { \
74  asm volatile( /*push lr on stack, to use it as a general purpose reg.*/ \
75  "stmfd sp!,{lr} \n\t" \
76  /*load ctxsave and dereference the pointer*/ \
77  "ldr lr,=ctxsave \n\t" \
78  "ldr lr,[lr] \n\t" \
79  /*save all thread registers except pc*/ \
80  "stmia lr,{r0-lr}^ \n\t" \
81  /*add a nop as required after stm ^ (read ARM reference about stm(2))*/ \
82  "nop \n\t" \
83  /*move lr to r0, restore original lr (return address) and save it*/ \
84  "add r0,lr,#60 \n\t" \
85  "ldmfd sp!,{lr} \n\t" \
86  "stmia r0!,{lr} \n\t" \
87  /*save spsr on top of ctxsave*/ \
88  "mrs r1,spsr \n\t" \
89  "stmia r0,{r1} \n\t"); \
90 }

112 #define restoreContext() \
113 { \
114  asm volatile( /*load ctxsave and dereference the pointer*/ \
115  /*also add 64 to make it point to "top of stack"*/ \
116  "ldr lr,=ctxsave \n\t" \
117  "ldr lr,[lr] \n\t" \
118  "add lr,lr,#64 \n\t" \
119  /*restore spsr*/ \
120  /*after this instructions, lr points to ctxsave[15] (return address)*/ \
121  "ldmda lr!,{r1} \n\t" \
122  "msr spsr,r1 \n\t" \
123  /*restore all thread registers except pc*/ \
124  "ldmdb lr,{r0-lr}^ \n\t" \
125  /*add a nop as required after ldm ^ (read ARM reference about ldm(2))*/ \
126  "nop \n\t" \
127  /*now that lr points to return address, return from interrupt*/ \
128  "ldr lr,[lr] \n\t" \
129  "movs pc,lr \n\t"); \
130 }
