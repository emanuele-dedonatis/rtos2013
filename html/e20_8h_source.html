193 template<unsigned SlotSize>
194 void FixedEventQueueBase<SlotSize>::postImpl(Callback<SlotSize>& event,
195  Callback<SlotSize> *events, unsigned int size)
196 {
197  //Not FastInterruptDisableLock as the operator= of the bound
198  //parameters of the Callback may allocate
199  InterruptDisableLock dLock;
200  while(n>=size)
201  {
202  WaitingList w;
203  w.token=false;
204  w.t=Thread::IRQgetCurrentThread();
205  w.next=waitingPut;
206  waitingPut=&w;
207  while(w.token==false)
208  {
209  Thread::IRQwait();
210  {
211  InterruptEnableLock eLock(dLock);
212  Thread::yield();
213  }
214  }
215  }
216  IRQpostImpl(event,events,size);
217 }

219 template<unsigned SlotSize>
220 bool FixedEventQueueBase<SlotSize>::IRQpostImpl(Callback<SlotSize>& event,
221  Callback<SlotSize> *events, unsigned int size, bool *hppw)
222 {
223  if(n>=size) return false;
224  events[put]=event; //This may allocate memory
225  if(++put>=size) put=0;
226  n++;
227  if(waitingGet)
228  {
229  Thread *t=Thread::IRQgetCurrentThread();
230  if(hppw && waitingGet->t->IRQgetPriority()>t->IRQgetPriority())
231  *hppw=true;
232  waitingGet->token=true;
233  waitingGet->t->IRQwakeup();
234  waitingGet=waitingGet->next;
235  }
236  return true;
237 }

239 template<unsigned SlotSize>
240 void FixedEventQueueBase<SlotSize>::runImpl(Callback<SlotSize> *events,
241  unsigned int size)
242 {
243  //Not FastInterruptDisableLock as the operator= of the bound
244  //parameters of the Callback may allocate
245  InterruptDisableLock dLock;
246  for(;;)
247  {
248  while(n<=0)
249  {
250  WaitingList w;
251  w.token=false;
252  w.t=Thread::IRQgetCurrentThread();
253  w.next=waitingGet;
254  waitingGet=&w;
255  while(w.token==false)
256  {
257  Thread::IRQwait();
258  {
259  InterruptEnableLock eLock(dLock);
260  Thread::yield();
261  }
262  }
263  }
264  Callback<SlotSize> f=events[get]; //This may allocate memory
265  if(++get>=size) get=0;
266  n--;
267  if(waitingPut)
268  {
269  waitingPut->token=true;
270  waitingPut->t->IRQwakeup();
271  waitingPut=waitingPut->next;
272  }
273  {
274  InterruptEnableLock eLock(dLock);
275  f();
276  }
277  }
