410 template<typename T>
411 intrusive_ref_ptr<T> intrusive_ref_ptr<T>::atomic_load() const
412 {
413  intrusive_ref_ptr<T> result; // This gets initialized with 0
414 
415  // According to the C++ standard, this causes undefined behaviour if
416  // T has virtual functions, but GCC (and clang) have an implementation
417  // based on a compiler intrinsic that does the right thing: it produces
418  // the correct answer together with a compiler warning for classes with
419  // virtual destructors, and even for classes using multiple inheritance.
420  // It (obviously) fails for classes making use of virtual inheritance, but
421  // in this case the warning is promoted to an error. The last thing to
422  // fix is to remove the warning using the #pragma
423  // As a result, intrusive_ref_ptr can't be used with classes that have
424  // virtual base classes, but that's an acceptable limitation, especially
425  // considering that you get a meaningful compiler error if accidentally
426  // trying to use it in such a case.
427  #pragma GCC diagnostic ignored "-Winvalid-offsetof"
428  const int offsetBytes=offsetof(T,intrusive.referenceCount);
429  #pragma GCC diagnostic pop
430 
431  // This is a static assert to check that referenceCount is
432  // properly aligned for the following code to work
433  typedef char check[(offsetBytes % sizeof(int))==0 ? 1 : -1];
434 
435  const int offsetInt=offsetBytes/sizeof(int);
436  void * const volatile * objectPtrAddr=
437  reinterpret_cast<void * const volatile *>(&object);
438  void *loadedObject=atomicFetchAndIncrement(objectPtrAddr,offsetInt,1);
439  // This does not increment referenceCount, as it was done before
440  result.object=reinterpret_cast<T*>(loadedObject);
441  return result;
442 }

444 template<typename T>
445 intrusive_ref_ptr<T> intrusive_ref_ptr<T>::atomic_exchange(
446  intrusive_ref_ptr<T>& r)
447 {
448  //Note: this is safe with respect to assignment to self
449  T *temp=r.object;
450  if(temp) atomicAdd(&temp->intrusive.referenceCount,1);
451 
452  // This is a static assert to check that the following reinterpret_casts
453  // will work as intended. This also means that this code won't work on
454  // 64bit machines but for Miosix this isn't a problem for now.
455  typedef char check[sizeof(void*)==sizeof(int) ? 1 : -1];
456 
457  int tempInt=reinterpret_cast<int>(temp);
458  volatile int *objectAddrInt=reinterpret_cast<volatile int*>(&object);
459  temp=reinterpret_cast<T*>(atomicSwap(objectAddrInt,tempInt));
460 
461  intrusive_ref_ptr<T> result; // This gets initialized with 0
462  // This does not increment referenceCount, as the pointer was swapped
463  result.object=temp;
464  return result;
465 }
