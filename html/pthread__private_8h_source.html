42 static inline void IRQdoMutexLock(pthread_mutex_t *mutex,
43  FastInterruptDisableLock& d)
44 {
45  void *p=reinterpret_cast<void*>(Thread::IRQgetCurrentThread());
46  if(mutex->owner==0)
47  {
48  mutex->owner=p;
49  return;
50  }
51 
52  //This check is very important. Without this attempting to lock the same
53  //mutex twice won't cause a deadlock because the Thread::IRQwait() is
54  //enclosed in a while(owner!=p) which is immeditely false.
55  if(mutex->owner==p)
56  {
57  if(mutex->recursive>=0)
58  {
59  mutex->recursive++;
60  return;
61  } else errorHandler(MUTEX_DEADLOCK); //Bad, deadlock
62  }
63 
64  WaitingList waiting; //Element of a linked list on stack
65  waiting.thread=p;
66  waiting.next=0; //Putting this thread last on the list (lifo policy)
67  if(mutex->first==0)
68  {
69  mutex->first=&waiting;
70  mutex->last=&waiting;
71  } else {
72  mutex->last->next=&waiting;
73  mutex->last=&waiting;
74  }
75 
76  //The while is necessary because some other thread might call wakeup()
77  //on this thread. So the thread can wakeup also for other reasons not
78  //related to the mutex becoming free
79  while(mutex->owner!=p)
80  {
81  Thread::IRQwait();//Returns immediately
82  {
83  FastInterruptEnableLock eLock(d);
84  Thread::yield(); //Now the IRQwait becomes effective
85  }
86  }
87 }
