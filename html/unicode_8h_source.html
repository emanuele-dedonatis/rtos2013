134 template<typename Iter>
135 char32_t Unicode::nextUtf8(Iter& it, Iter end, bool checkEnd)
136 {
137  //End of string at the beginning, return 0
138  if(checkEnd && it==end) return 0;
139 
140  //Note: cast to unsigned char to prevent sign extension if *it > 0x7f
141  char32_t c=static_cast<unsigned char>(*it++);
142 
143  //Common case first: ASCII
144  if(c<0x80) return c;
145 
146  //If not ASCII, decode to utf32
147  int additionalBytes;
148  if((c & 0xe0)==0xc0) { c &= 0x1f; additionalBytes=1; } //110xxxxx
149  else if((c & 0xf0)==0xe0) { c &= 0x0f; additionalBytes=2; } //1110xxxx
150  else if((c & 0xf8)==0xf0) { c &= 0x07; additionalBytes=3; } //11110xxx
151  else return invalid;
152  for(int i=0;i<additionalBytes;i++)
153  {
154  //End of string in the middle of a char, return invalid
155  if(checkEnd && it==end) return invalid;
156  char32_t next=static_cast<unsigned char>(*it++);
157  //This includes the case next==0
158  if((next & 0xc0)!=0x80) return invalid;
159  c<<=6;
160  c |= next & 0x3f;
161  }
162  //Detect overlong encodings as errors to prevent vulnerabilities
163  switch(additionalBytes)
164  {
165  case 1:
166  if(c<0x80) return invalid;
167  break;
168  case 2:
169  if(c<0x800) return invalid;
170  break;
171  case 3:
172  if(c<0x10000) return invalid;
173  break;
174  }
175 
176  //Reserved space for surrogate pairs in utf16 are invalid code points
177  if(c>=0xd800 && c<= 0xdfff) return invalid;
178  //Unicode is limited in the range 0-0x10ffff
179  if(c>0x10ffff) return invalid;
180  return c;
181 }
